// Autenticación //
///////////////////

Login
-----

Nombre del archivo: loginXXYYYY.xml
	XX: número de grupo
	YYYY: número de terminal virtual
    
Descripción del xml:
<?xml version="1.0" encoding="UTF-8" ?>
<login vTerm="id terminal virtual" password="password de administrador"/>


RespuestaLogin (aceptar/denegar Login)
--------------

Nombre del archivo: respuestaLoginXXYYYY.xml
	XX: número de grupo
	YYYY: número de terminal virtual
    
Descripción del xml:
<?xml version="1.0" encoding="UTF-8" ?>
<respuestaLogin vTerm="id terminal virtual">
    <aceptado fueAceptado="si | no" />
    <descripcion>"información adicional, de por qué no fue aceptado por ejemplo u otra informacion relevante"</descripcion>
</respuestaLogin>

Configurar ModoDirigido::Tragamonedas
-------------------------------------

Nombre del archivo: configurarModoDirigidoTragamonedasXXYYYY.xml
	XX: número de grupo
	YYYY: número de terminal virtual
    
Descripción del xml:
<?xml version="1.0" encoding="UTF-8" ?>
<configurarModoDirigidoTragamonedas vTerm="id terminal virtual">
    <controlResultados activo="si | no">
        <resultado>
            [<rueda1>"cereza" | "bar simple" | "bar doble" | "bar triple" | "dinosaurio"</rueda1>
            <rueda2>"cereza" | "bar simple" | "bar doble" | "bar triple" | "dinosaurio"</rueda2>
            <rueda3>"cereza" | "bar simple" | "bar doble" | "bar triple" | "dinosaurio"</rueda3>]
        </resultado>
    </controlResultados>
    <controlTipoJugadas activo="si | no">
        <tipo>"Normal" | "TodosPonen" | "NoNormal" | "NoTodosPonen" | "NoFeliz"</tipo>
    </controlTipoJugadas>
</configurarModoDirigidoTragamonedas>
